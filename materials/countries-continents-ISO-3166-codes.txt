  * Country names, continent names and ISO-3166 codes.
  */
CREATE TABLE CONTINENTS (
  CODE CHAR(2) NOT NULL, --'Continent code',
  CONTINENT_NAME VARCHAR(255),

-- Create Countries table
CREATE TABLE COUNTRIES (
  CODE CHAR(2) NOT NULL, --'Two-letter country code (ISO 3166-1 alpha-2)',
  COUNTRY_NAME VARCHAR(255) NOT NULL, --'English country name',
  FULL_NAME VARCHAR(255) NOT NULL, --'Full English country name',
  ISO3 CHAR(3) NOT NULL, --'Three-letter country code (ISO 3166-1 alpha-3)',
  COUNTRY_NUMBER NUMBER(3) NOT NULL, --'Three-digit country number (ISO 3166-1 numeric)',
  CONTINENT_CODE CHAR(2) NOT NULL,
  PRIMARY KEY (CODE),
  --KEY CONTINENT_CODE (CONTINENT_CODE),
  CONSTRAINT fk_countries_continents FOREIGN KEY (CONTINENT_CODE) REFERENCES CONTINENTS (CODE)
